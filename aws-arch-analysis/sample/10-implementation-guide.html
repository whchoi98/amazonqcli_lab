2.1 EC2 Right-sizing 구현

# EC2 사용률 분석 스크립트
import boto3
import pandas as pd
from datetime import datetime, timedelta

def analyze_ec2_utilization():
    ec2 = boto3.client('ec2')
    cloudwatch = boto3.client('cloudwatch')

    instances = ec2.describe_instances()
    recommendations = []

    for reservation in instances['Reservations']:
        for instance in reservation['Instances']:
            # CPU 사용률 분석
            metrics = cloudwatch.get_metric_statistics(
                Namespace='AWS/EC2',
                MetricName='CPUUtilization',
                Dimensions=[
                    {'Name': 'InstanceId', 'Value': instance['InstanceId']}
                ],
                StartTime=datetime.utcnow() - timedelta(days=14),
                EndTime=datetime.utcnow(),
                Period=3600,
                Statistics=['Average']
            )

            avg_cpu = sum([m['Average'] for m in metrics['Datapoints']]) / len(metrics['Datapoints'])

            if avg_cpu < 20:
                recommendations.append({
                    'InstanceId': instance['InstanceId'],
                    'CurrentType': instance['InstanceType'],
                    'AvgCPU': avg_cpu,
                    'Recommendation': 'Downsize or schedule'
                })

    return recommendations

# Lambda 함수 예제: 이벤트 처리
import json
import boto3

def lambda_handler(event, context):
    """
    S3 이벤트를 처리하는 Lambda 함수
    """
    s3_client = boto3.client('s3')
    sns_client = boto3.client('sns')

    for record in event['Records']:
        bucket = record['s3']['bucket']['name']
        key = record['s3']['object']['key']

        # 파일 처리 로직
        try:
            response = s3_client.get_object(Bucket=bucket, Key=key)
            # 처리 로직 구현

            # 성공 알림
            sns_client.publish(
                TopicArn='arn:aws:sns:region:account:success-topic',
                Message=f'Successfully processed {key}',
                Subject='File Processing Success'
            )

        except Exception as e:
            # 오류 알림
            sns_client.publish(
                TopicArn='arn:aws:sns:region:account:error-topic',
                Message=f'Error processing {key}: {str(e)}',
                Subject='File Processing Error'
            )

    return {
        'statusCode': 200,
        'body': json.dumps('Processing completed')
    }

# security-audit.py
import boto3
import json

def audit_security_groups():
    """보안 그룹 감사"""
    ec2 = boto3.client('ec2')

    security_groups = ec2.describe_security_groups()
    risky_groups = []

    for sg in security_groups['SecurityGroups']:
        for rule in sg['IpPermissions']:
            for ip_range in rule.get('IpRanges', []):
                if ip_range.get('CidrIp') == '0.0.0.0/0':
                    risky_groups.append({
                        'GroupId': sg['GroupId'],
                        'GroupName': sg['GroupName'],
                        'Protocol': rule.get('IpProtocol'),
                        'Port': rule.get('FromPort')
                    })

    return risky_groups

def audit_s3_buckets():
    """S3 버킷 보안 감사"""
    s3 = boto3.client('s3')

    buckets = s3.list_buckets()
    public_buckets = []

    for bucket in buckets['Buckets']:
        try:
            acl = s3.get_bucket_acl(Bucket=bucket['Name'])
            for grant in acl['Grants']:
                if grant['Grantee'].get('URI') == 'http://acs.amazonaws.com/groups/global/AllUsers':
                    public_buckets.append(bucket['Name'])
        except:
            continue

    return public_buckets

if __name__ == "__main__":
    print("Security Audit Results:")
    print("Risky Security Groups:", audit_security_groups())
    print("Public S3 Buckets:", audit_s3_buckets())

# performance-dashboard.py
import boto3
import matplotlib.pyplot as plt
from datetime import datetime, timedelta

def generate_cost_report():
    """비용 절약 효과 리포트 생성"""
    ce = boto3.client('ce')

    end_date = datetime.now().strftime('%Y-%m-%d')
    start_date = (datetime.now() - timedelta(days=30)).strftime('%Y-%m-%d')

    response = ce.get_cost_and_usage(
        TimePeriod={
            'Start': start_date,
            'End': end_date
        },
        Granularity='MONTHLY',
        Metrics=['BlendedCost'],
        GroupBy=[
            {
                'Type': 'DIMENSION',
                'Key': 'SERVICE'
            }
        ]
    )

    # 비용 데이터 시각화
    services = []
    costs = []

    for result in response['ResultsByTime'][0]['Groups']:
        service = result['Keys'][0]
        cost = float(result['Metrics']['BlendedCost']['Amount'])
        services.append(service)
        costs.append(cost)

    plt.figure(figsize=(12, 8))
    plt.pie(costs, labels=services, autopct='%1.1f%%')
    plt.title('AWS Cost Distribution')
    plt.savefig('cost-distribution.png')
    plt.show()
